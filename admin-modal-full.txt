19:   isOpen,
20:   onClose,
21:   selectedTableNumber,

23:   currentTable,

25: }: AdminModalProps) {

30:

34:   const [targetNumberInput, setTargetNumberInput] = useState('');
35:   const [resetNumberInput, setResetNumberInput] = useState('');
36:

40:   const [isLoading, setIsLoading] = useState(false);
41: 
42:   // Sync state date when table changes
43:   useEffect(() => {
44:     if (currentTable) {
45:       setDrawDate(currentTable.draw_date || '');
46:       setTargetTableInput(String(currentTable.table_number));

49:

52:     if (isUnlocked && isOpen) {
53:       fetchAdminStats();
54:     }
55:   }, [isUnlocked, isOpen, tableReservations]);
56: 
57:   const fetchAdminStats = async () => {

59:       const data = await db.getStats();

62:       console.error('Stats fetch error:', err);
63:     }
64:   };
65: 
66:   const clearMessages = () => {

68:   };
69:

86:       setPinError('Unable to authenticate PIN.');
87:     } finally {
88:       setIsLoading(false);
89:     }
90:   };
91: 
92:   const handleLogout = () => {
93:     setIsUnlocked(false);
94:     setPinError(null);
95:     setAdminMsg(null);
96:   };
97: 
98:   // 2. Save Draw Date code
99:   const handleSaveDrawDate = async () => {
100:     clearMessages();
101:     const tNum = parseInt(targetTableInput);
102:     if (isNaN(tNum) || tNum < 1 || tNum > 10) {
103:       setAdminMsg({ text: 'Valid table number (1-10) is required.', isError: true });
104:       return;
105:     }
106:

108:       setIsLoading(true);
109:       const formattedDate = drawDate ? drawDate : null;
110:       const success = await db.setTableDrawDate(tNum, formattedDate);
111:       if (success) {
112:         setAdminMsg({ text: `Draw date saved successfully for Table ${tNum}.` });
113:         onRefreshData();

115:         setAdminMsg({ text: 'Failed to update draw date.', isError: true });
116:       }
117:     } catch (err: any) {
118:       setAdminMsg({ text: err.message || 'Error saving draw date.', isError: true });
119:     } finally {
120:       setIsLoading(false);
121:     }
122:   };
123: 
124:   // 3. Open or Lock table code
125:   const handleSetTableLock = async (isOpenState: boolean) => {
126:     clearMessages();
127:     const tNum = parseInt(targetTableInput);
128:     if (isNaN(tNum) || tNum < 1 || tNum > 10) {
129:       setAdminMsg({ text: 'Valid table number (1-10) is required.', isError: true });
130:       return;
131:     }
132: 
133:     try {
134:       setIsLoading(true);
135:       const success = await db.setTableStatus(tNum, isOpenState);
136:       if (success) {
137:         setAdminMsg({ text: `Table ${tNum} is now ${isOpenState ? 'OPEN' : 'LOCKED'}.` });

140:         setAdminMsg({ text: `Failed to set Table ${tNum} lock status.`, isError: true });
141:       }
142:     } catch (err: any) {
143:       setAdminMsg({ text: err.message || 'Error updating table status.', isError: true });
144:     } finally {
145:       setIsLoading(false);
146:     }
147:   };
148: 
149:   // 4. Confirm Paid Number action
150:   const handleConfirmPayment = async () => {
151:     clearMessages();
152:     const cellNum = parseInt(targetNumberInput);
153:     if (isNaN(cellNum) || cellNum < 1 || cellNum > 42) {
154:       setAdminMsg({ text: 'Valid number (1-42) is required to confirm.', isError: true });
155:       return;
156:     }
157: 
158:     try {
159:       setIsLoading(true);
160:       const res = await db.confirmPayment(selectedTableNumber, cellNum);
161:       if (res.success) {
162:         luxuryAudio.praiseGrandBell();
163:         setAdminMsg({ text: `Number ${cellNum} payment confirmed inside Table ${selectedTableNumber}.` });
164:         setTargetNumberInput('');

167:         setAdminMsg({ text: res.error || 'Failed to confirm payment.', isError: true });
168:       }
169:     } catch (err: any) {
170:       setAdminMsg({ text: err.message || 'Error confirming payment.', isError: true });
171:     } finally {
172:       setIsLoading(false);
173:     }
174:   };
175: 
176:   // 5. Reset one Number action with prompt double-checks
177:   const handleResetNumber = async () => {
178:     clearMessages();
179:     const cellNum = parseInt(resetNumberInput);
180:     if (isNaN(cellNum) || cellNum < 1 || cellNum > 42) {
181:       setAdminMsg({ text: 'Valid number (1-42) is required to reset.', isError: true });
182:       return;
183:     }
184: 
185:     const conf = window.confirm(`Are you sure you want to reset and clear slot #${cellNum} entirely?`);
186:     if (!conf) return;
187: 
188:     try {
189:       setIsLoading(true);
190:       const res = await db.resetNumber(selectedTableNumber, cellNum);
191:       if (res.success) {
192:         setAdminMsg({ text: `Number ${cellNum} reset successfully.` });
193:         setResetNumberInput('');
194:         onRefreshData();
195:       } else {
196:         setAdminMsg({ text: res.error || 'Failed to reset number.', isError: true });
197:       }
198:     } catch (err: any) {
199:       setAdminMsg({ text: err.message || 'Error resetting number.', isError: true });
200:     } finally {
201:       setIsLoading(false);
202:     }
203:   };
204: 
205:   // 6. Danger Area Clear All Numbers action with literal verification check
206:   const handleClearAllNumbers = async () => {
207:     clearMessages();

209:       `DANGER: This clears ALL active reservations from Table ${selectedTableNumber}!\nProfit history is kept.\nTo proceed, type exactly "CLEAR" below:`

212:     if (response !== 'CLEAR') {
213:       setAdminMsg({ text: 'Clear action cancelled. Verification text mismatched.', isError: true });

215:     }

217:     try {
218:       setIsLoading(true);

220:       if (success) {
221:         setAdminMsg({ text: `All numbers for Table ${selectedTableNumber} have been cleared.` });

223:       } else {
224:         setAdminMsg({ text: 'Failed to clear all numbers.', isError: true });
225:       }
226:     } catch (err: any) {
227:       setAdminMsg({ text: err.message || 'Error clearing numbers.', isError: true });
228:     } finally {
229:       setIsLoading(false);
230:     }
231:   };
232: 
233:   // 7. Export current table reservations to CSV
234:   const handleExportCSV = () => {
235:     if (tableReservations.length === 0) {
236:       setAdminMsg({ text: `No reservations to export for Table ${selectedTableNumber}.`, isError: true });
237:       return;
238:     }
239: 
240:     try {
241:       // Sort by chosen table slot number
242:       const sortedResult = [...tableReservations].sort((a, b) => a.number - b.number);
243:       const headers = ['Table Number', 'Slot Number', 'Name', 'Phone', 'Status', 'Price Paid ($)', 'Reserved At', 'Paid At'];
244:       
245:       const rows = sortedResult.map(res => [
246:         res.table_number,
247:         res.number,
248:         res.name,
249:         res.phone || '',
250:         res.status.toUpperCase(),
251:         res.price,

256:       const csvContent = [
257:         headers.join(','),

264:             return str;
265:           }).join(',')

268: 
269:       const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });

273:       link.setAttribute('download', `table_${selectedTableNumber}_reservations_${new Date().toISOString().split('T')[0]}.csv`);
274:       document.body.appendChild(link);
275:       link.click();
276:       document.body.removeChild(link);
277:       URL.revokeObjectURL(url);
278:       
279:       setAdminMsg({ text: `Exported ${tableReservations.length} reservations from Table ${selectedTableNumber} to CSV!` });
280:     } catch (err: any) {
281:       setAdminMsg({ text: err.message || 'Error exporting CSV.', isError: true });
282:     }
283:   };
284: 
285:   // Render logic
286:   if (!isOpen) return null;
287: 
288:   return (
289:     <div className="fixed inset-0 bg-black/90 backdrop-blur-md flex items-center justify-center p-4 z-50 overflow-y-auto">
290:       <div className="w-full max-w-xl bg-[#050403] border-4 border-gold rounded-none shadow-[0_10px_50px_rgba(212,175,55,0.25)] flex flex-col max-h-[90vh] overflow-hidden custom-shadow">
291:         
292:         {/* Header Ribbon */}
293:         <div className="flex items-center justify-between p-4 border-b border-gold/20 bg-[#070605]">

297:               {isUnlocked ? 'ADMIN PANEL' : 'ADMIN ACCESS'}
298:             </h3>
299:           </div>
300:           <button
301:             onClick={onClose}
302:             className="p-1 text-stone-400 hover:text-gold hover:bg-[#0a0703] border border-transparent hover:border-gold/30 rounded-none transition-all cursor-pointer"

304:             <X className="w-5 h-5" />
305:           </button>

308:         {/* Scrollable Modal Content */}

316:                   <Lock className="w-8 h-8" />
317:                 </div>
318:                 <p className="text-xs text-stone-400 leading-relaxed font-sans uppercase tracking-[0.2em]">
319:                   Pin Code Required
320:                 </p>
321:               </div>
322: 
323:               <div>
324:                 <label className="block text-[10px] tracking-wider text-stone-400 font-sans uppercase mb-1.5 text-center font-bold">
325:                   PIN CODE
326:                 </label>
327:                 <input
328:                   type="password"
329:                   value={pin}
330:                   onChange={(e) => setPin(e.target.value)}
331:                   placeholder="â€¢â€¢â€¢â€¢"
332:                   className="w-full bg-[#0a0703] text-center text-xl font-serif tracking-[1em] text-gold-light py-3 rounded-none border border-gold focus:outline-2 focus:outline-gold-light placeholder-stone-700"
333:                   disabled={isLoading}
334:                   autoFocus
335:                 />
336:               </div>

338:               {pinError && (

340:                   {pinError}
341:                 </p>

344:               <button
345:                 type="submit"
346:                 disabled={isLoading}

384:                 <div className="grid grid-cols-2 gap-4">

388:                       TABLE NUMBER

390:                     <select

395:                       {Array.from({ length: 10 }, (_, i) => (
396:                         <option key={i + 1} value={i + 1}>
397:                           Table {i + 1}
398:                         </option>
399:                       ))}
400:                     </select>

403:                   {/* Draw date setter */}
